class User {
    - id : string
    - nama : string
    - email : string
    - password : string
    - role : string (enum: admin, collector, recycler)
    + register() : void
    + login() : boolean
    + logout()
    + updateProfil() : void
    + deleteUser() : void
}

class Admin {
    + manageUser() : void
    + viewAllUser() : void
}

class Collector {
    + addRequest() : void
    + viewRequest() : void
}

class Recycler {
    + addRequest() : void
    + viewRequest() : void
}

class WasteItems {
    - id : int
    - type[] : string
    - weight : float
    - pricePerKg : float
    + calculatePrice() : float
}

class OrganicWaste {
    - decompositionTime : datetime
    + isCompostable() : bool
}

class PlasticWaste {
    - recyclabilityGrade : string
    + calculateRecyclingCost() : float
}

class MetalWaste {
    - metalType : string
    + isValuable() : boolean
}

class CollectionRequest {
    - id : int
    - userId : int
    - pickUpDate : datetime
    - status : string (enum: Pending, InProgress, Complete)
    + addRequest() : void
    + updateRequest() : void
}

class Transactions {
    - id : int
    - userId : int
    - totalAmount : float
    - date : datetime
    + generateReceipt() : string
    + viewHistory() : array
}

class Report {
    - id : int
    - userId : int
    - month : string
    - totalCollected : float
    + generateMonthlyReport() : void
    + viewReport() : void
}

class Notification {
    - id : int
    - userId : int
    - message : string
    - createdAt : datetime
    + sendNotification() : void
}
